安装g2o
// 从github上下载源码
$ https://github.com/RainerKuemmerle/g2o/tree/20170730_git

// 安装依赖
$ sudo apt-get install cmake libeigen3-dev libsuitesparse-dev qtdeclarative5-dev qt5-qmake libqglviewer-dev


// 编译
$ cd g2o
$ mkdir build
$ cd build
$ sudo ldconfig
$ cmake ..
$ make -j4

//安装
$ sudo make install



安装gflags
// 克隆源码
$ git clone https://github.com/gflags/gflags

// 解压并进入文件夹
$ cd gflags

// 编译及安装
$ mkdir build
$ cd build
$ cmake ..
$ make -j4
$ sudo make install


安装glog
// 克隆源码
$ git clone https://github.com/boboxxd/glog.git

// 解压并进入文件夹
$ cd glog

// 安装依赖
$ sudo apt-get install autoconf automake libtool

// 配置
$ ./autogen.sh
$ ./configure

// 编译及安装
$ mkdir build
$ cd build
$ cmake ..
$ make -j4
$ sudo make install
